Python

 1def fibonacci(n):
 2    """Generate Fibonacci sequence up to n terms."""
 3    a, b = 0, 1
 4    result = []
 5    for _ in range(n):
 6        result.append(a)
 7        a, b = b, a + b
 8    return result
 9
10# Generate first 10 Fibonacci numbers
11print(fibonacci(10))

 5unsigned long factorial(int n) {
 6    if (n <= 1) {
 7        return 1;
 8    }
